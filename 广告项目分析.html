# -*- coding: utf-8 -*-
import pandas as pd
import numpy as np 

# 定义文件路径
file_path = r"C:\Users\lili.li\Downloads\产品报表_2025-04-18-2025-07-16_分日 (1).xlsx"

# 加载数据
try:
    df = pd.read_excel(file_path)
    print("数据加载成功！")
except Exception as e:
    print(f"加载数据时发生错误：{e}")
    exit()

print("\n--- 第二步：数据清洗与预处理 ---")

# 1. 
column_mapping = {
    '产品名称': 'ProductName',
    '产品ID': 'ProductID',
    '时间': 'Date',
    '花费': 'Spend',
    '曝光量': 'Impressions',
    '点击量': 'Clicks',
    '下载量': 'Downloads',
    '激活量': 'Activations',
    '收益': 'Revenue'
}

df.rename(columns=column_mapping, inplace=True)
print("\n列名标准化完成。当前列名：", df.columns.tolist())


# 2. 日期列处理
# 将 'Date' 列转换为日期时间格式
df['Date'] = pd.to_datetime(df['Date'], errors='coerce') # errors='coerce' 将无效解析为 NaT

# 检查是否存在无效日期（NaT），并决定如何处理。根据你的df.info()，目前没有缺失值，所以可能没有NaT
if df['Date'].isnull().any():
    print("警告：'Date' 列存在无效日期，已转换为NaT。")
    # 如果有NaT，可以选择删除这些行：
    # df.dropna(subset=['Date'], inplace=True)
    # 或者用某个默认日期填充，这取决于业务需求
else:
    print("日期列处理完成，无无效日期。")


# 3. 核心指标的数值类型校验（根据df.info()，这些已经是正确类型，但为了稳健性可以保留）
# Spend, Activations, Revenue 已经是 float64 或 int64，不需要强制转换
# 但为了计算的稳健性，确保关键列没有非数值
numeric_cols_to_check = ['Spend', 'Activations', 'Revenue', 'Impressions', 'Clicks', 'Downloads']
for col in numeric_cols_to_check:
    df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0) # 转换为数值，非数值设为0

print("\n核心指标数值类型校验完成，并填充了可能存在的非数值为0。")


# 4. 计算核心指标：激活成本 (CPA) 和 ROAS
# 激活成本 (CPA) = 花费 / 激活量
# ROAS = 收益 / 花费

# 计算 CPA：避免除以零，当 Activations 为 0 时，CPA 为 NaN。
df['CPA'] = df.apply(lambda row: row['Spend'] / row['Activations'] if row['Activations'] > 0 else np.nan, axis=1)
# 对于极高的 CPA (比如激活量极小，花费很大)，或无限大值，可以考虑填充。这里我们用np.nan，后续可视化时可以灵活处理。
df['CPA'].replace([np.inf, -np.inf], np.nan, inplace=True) # 替换正负无穷大值
# 考虑到CPA是成本，当激活量为0但有花费时，其成本实际是无限高。这里将NaN填充为Spend，意味着没有激活的成本就是全部花费。
# 或者，可以填充一个很大的固定值，或者在分析时单独处理NaN。这里选择填充 Spend。
df['CPA'].fillna(df['Spend'], inplace=True) # 如果没有激活（Activations=0），CPA就是花费本身

# 计算 ROAS：避免除以零，当 Spend 为 0 时，ROAS 为 NaN。
df['ROAS'] = df.apply(lambda row: row['Revenue'] / row['Spend'] if row['Spend'] > 0 else np.nan, axis=1)
df['ROAS'].replace([np.inf, -np.inf], np.nan, inplace=True) # 替换正负无穷大值
# 当 Spend 为 0 时，ROAS 无意义，填充为 0。因为没有花费就没有回报。
df['ROAS'].fillna(0, inplace=True) # 如果没有花费（Spend=0），ROAS就是0

print("\nCPA 和 ROAS 已计算。")

# 5. 再次查看处理后的数据概况
print("\n数据清洗与预处理后，数据前5行：")
print(df.head())

print("\n数据清洗与预处理后，数据基本信息：")
df.info()

print("\n数据清洗与预处理后，核心指标统计摘要：")
print(df[['Spend', 'Activations', 'Revenue', 'CPA', 'ROAS']].describe())

# 保存清洗后的数据到新的CSV文件，方便后续步骤直接读取
df.to_csv("cleaned_ad_performance_data.csv", index=False, encoding='utf-8-sig') # 使用utf-8-sig避免中文乱码
print("\n清洗后的数据已保存到 'cleaned_ad_performance_data.csv'")
